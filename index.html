    CONSTRAINT PK_Status PRIMARY KEY (StatusId)
);

CREATE TABLE [Product] (
    ProductId      INT           NOT NULL IDENTITY(1,1),
    ProductArticul NVARCHAR(100) NOT NULL,
    ProductName    NVARCHAR(MAX) NOT NULL,
    ProductUnit    NVARCHAR(MAX) NOT NULL,
    ProductPrice   DECIMAL(10,2) NOT NULL,
    SupplierId     INT           NOT NULL,
    ManufacturerId INT           NOT NULL,
    CategoryId     INT           NOT NULL,
    Discount       INT           NOT NULL,
    CountInStock   INT           NOT NULL,
    Description    NVARCHAR(MAX) NULL,
    Photopath      NVARCHAR(MAX) NULL,
    CONSTRAINT PK_Product              PRIMARY KEY (ProductArticul),
    CONSTRAINT FK_Product_Supplier     FOREIGN KEY (SupplierId)     REFERENCES [Supplier](SupplierId),
    CONSTRAINT FK_Product_Manufacturer FOREIGN KEY (ManufacturerId) REFERENCES [Manufacturer](ManufacturerId),
    CONSTRAINT FK_Product_Category     FOREIGN KEY (CategoryId)     REFERENCES [Category](CategoryId)
);

CREATE TABLE [Order] (
    OrderId     INT  NOT NULL IDENTITY(1,1),

    Cod         INT  NOT NULL,
    StatusId    INT  NOT NULL,
    CONSTRAINT PK_Order           PRIMARY KEY (OrderId),
    CONSTRAINT FK_Order_PickPoint FOREIGN KEY (PickPointId) REFERENCES [PickPoint](PickPointId),
    CONSTRAINT FK_Order_User      FOREIGN KEY (UserId)      REFERENCES [User](UserId),
    CONSTRAINT FK_Order_Status    FOREIGN KEY (StatusId)    REFERENCES [Status](StatusId)

CREATE TABLE [OrderProduct] (

Как зайти в БД

SSMS → «Соединить» → «Свойства подключения». Заполните как на скриншоте:

  • Имя сервера: 192.168.58.69
  • Проверка подлинности: SQL Server
  • Имя пользователя: user10
  • Пароль: ваш пароль
  • Шифровать: Обязательно
  • Сертификат сервера: включить «Доверять сертификату сервера»
  • База данных: user10

После настройки нажмите «Проверить подключение», затем «ОК».

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace FirstAttempt
{
    public partial class Product
    {
        public string TypeAndName
        {
            get => $"{Category.CategoryName} | {ProductName}";
        }

        public string Background
        {
            get
            {
                if (this.Discount > 15)
                    return "#2E8B57";
                if (this.CountInStock <= 0)
                    return "#FF17F1EC";
                return "#fff";
            }
        }

        public string PriceColor
        {
            get
            {
                if (this.Discount > 0)
                    return "#FFEA0D0D";
                return "#FF060606";
            }
        }

        public string PriceTextDecorations
        {
            get
            {
                if (this.Discount > 0)
                    return "Strikethrough";
                return "None";
            }
        }

        public string CostWithDiscount
        {
            get
            {
                if (this.Discount > 0)
                {
                    var costTotal = Convert.ToDouble(ProductPrice) - Convert.ToDouble(ProductPrice) * Convert.ToDouble(Discount / 100.0);
                    return costTotal.ToString();
                }
                return "";
            }
        }
    }
}

using System.Windows.Data;
using System.Windows.Documents;
using System.Windows.Input;
using System.Windows.Media;
using System.Windows.Media.Imaging;
using System.Windows.Navigation;
using System.Windows.Shapes;

namespace FirstAttempt
{
    /// <summary>
    /// Interaction logic for Autho.xaml
    /// </summary>
    public partial class Autho : Page
    {
        public Autho()
        {
            InitializeComponent();
        }

        private void Button_Click(object sender, RoutedEventArgs e)
        {
            NavigationService.Navigate(new ClientPage(null));
        }

        private void Button_Click_1(object sender, RoutedEventArgs e)
        {
            var user = Entities.getContext().User.FirstOrDefault(u => u.Login == tbLogin.Text
                                                                && u.Password == pbPassword.Password);

            if (user == null)
            {
                MessageBox.Show("Такого пользователя не существует!", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
                return;
            }

            switch (user.RoleId)
            {
                // Admin
                case 1:
                    NavigationService.Navigate(new AdminPage(user));
                    break;
                // Manager
                case 2:
                    NavigationService.Navigate(new ManagerPage(user));
                    break;
                // Client
                case 3:
                    NavigationService.Navigate(new ClientPage(user));
                    break;
                default:
                    MessageBox.Show("Неверный логин или пароль!", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
                    break;
            }
        }
    }
}

ManagerPage.xaml.cs

using System.Linq;
using System.Windows;
using System.Windows.Controls;
using System.Windows.Input;

namespace FirstAttempt
{
    public partial class ManagerPage : Page
    {
        Entities entities = new Entities();

        public ManagerPage(User u)
        {
            InitializeComponent();
            LV.ItemsSource = Entities.getContext().Product.ToList();
            if (u != null)
                txtName.Content = u.FirstName + " " + u.LastName;

            comboSort.Items.Add("Без сортировки");
            comboSort.Items.Add("Количество по возрастанию");
            comboSort.Items.Add("Количество по убыванию");

            comboFiltr.Items.Add("Все поставщики");
            comboFiltr.Items.Add("Kari");
            comboFiltr.Items.Add("Обувь для вас");
        }

        private void tbSeacrh_TextChanged(object sender, TextChangedEventArgs e)
        {
            UpdateData();
        }

        private void comboSort_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            UpdateData();
        }

        private void comboFiltr_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            UpdateData();
        }

        private void UpdateData()
        {
            var result = entities.Product.ToList();

            if (comboSort.SelectedIndex == 1)
                result = result.OrderBy(p => p.CountInStock).ToList();
            if (comboSort.SelectedIndex == 2)
                result = result.OrderByDescending(p => p.CountInStock).ToList();

            result = result.Where(p =>
                p.ProductName.ToLower().Contains(tbSeacrh.Text.ToLower()) == true ||
                p.Description.ToLower().Contains(tbSeacrh.Text.ToLower()) == true
            ).ToList();

            if (comboFiltr.SelectedIndex == 1)
                result = result.Where(p => p.Supplier.SupplierName == "Kari").ToList();
            if (comboFiltr.SelectedIndex == 2)
                result = result.Where(p => p.Supplier.SupplierName == "Обувь для вас").ToList();

            LV.ItemsSource = result;
        }

        private void LV_MouseDown(object sender, MouseButtonEventArgs e)
        {
            NavigationService.Navigate(new AddEditPage(LV.SelectedItem as Product));
        }

        private void Page_IsVisibleChanged(object sender, DependencyPropertyChangedEventArgs e)
        {
            if (Visibility == Visibility.Visible)
                LV.ItemsSource = Entities.getContext().Product.ToList();
        }
    }
}

            comboFiltr.Items.Add("Kari");
            comboFiltr.Items.Add("Обувь для вас");

        private void tbSeacrh_TextChanged(object sender, TextChangedEventArgs e)
        {
            UpdateData();
        }

        private void comboSort_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            UpdateData();
        }

        private void comboFiltr_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            UpdateData();
        }

        private void UpdateData()
        {
            var result = entities.Product.ToList();

            if (comboSort.SelectedIndex == 1)
                result = result.OrderBy(p => p.CountInStock).ToList();
            if (comboSort.SelectedIndex == 2)
                result = result.OrderByDescending(p => p.CountInStock).ToList();

            result = result.Where(p =>
                p.ProductName.ToLower().Contains(tbSeacrh.Text.ToLower()) == true ||
                p.Description.ToLower().Contains(tbSeacrh.Text.ToLower()) == true
            ).ToList();

            if (comboFiltr.SelectedIndex == 1)
                result = result.Where(p => p.Supplier.SupplierName == "Kari").ToList();
            if (comboFiltr.SelectedIndex == 2)
                result = result.Where(p => p.Supplier.SupplierName == "Обувь для вас").ToList();

            LV.ItemsSource = result;
        }

        private void Button_Add(object sender, RoutedEventArgs e)
        {
            NavigationService.Navigate(new AddEditPage(null));
        }

        private void LV_MouseDown(object sender, MouseButtonEventArgs e)
        {
            NavigationService.Navigate(new AddEditPage(LV.SelectedItem as Product));
        }

        private void Page_IsVisibleChanged(object sender, DependencyPropertyChangedEventArgs e)
        {
            if (Visibility == Visibility.Visible)
                LV.ItemsSource = Entities.getContext().Product.ToList();
        }

        private void Button_Delete(object sender, RoutedEventArgs e)
        {
            var del_prod = LV.SelectedItem as Product;
            if (del_prod == null)
                return;

            if (MessageBox.Show("Вы уверены?", "Подтверждение", MessageBoxButton.YesNo, MessageBoxImage.Question) != MessageBoxResult.Yes)
                return;

            try
            {
                var exist = Entities.getContext().OrderProduct.First(p => p.ProductArt == del_prod.ProductArticul);
                MessageBox.Show("нельзя удалить. есть связь", "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
            }
            catch
            {
                Entities.getContext().Product.Remove(del_prod);
                Entities.getContext().SaveChanges();
                MessageBox.Show("Данные удалены", "Информация", MessageBoxButton.OK, MessageBoxImage.Information);
                LV.ItemsSource = Entities.getContext().Product.ToList();
            }
        }
    }

using System;
using System.Linq;
using System.Text;
using System.Windows;
using System.Windows.Controls;
using System.Windows.Media.Imaging;
using Microsoft.Win32;

namespace FirstAttempt
{
    public partial class AddEditPage : Page
    {
        Entities entities = new Entities();
        Product product = new Product();

        public AddEditPage(Product p)
        {
            InitializeComponent();
            comboCateg.ItemsSource = Entities.getContext().Category.ToList();
            comboManufact.ItemsSource = Entities.getContext().Manufacturer.ToList();
            comboSapplier.ItemsSource = Entities.getContext().Supplier.ToList();

            if (p != null)
            {
                product = p;
                tbArt.Visibility = Visibility.Visible;
                txtArt.Visibility = Visibility.Visible;
                txtArt.IsEnabled = false;
            }

            DataContext = product;
        }

        private void btnEnterImg_Click(object sender, RoutedEventArgs e)
        {
            OpenFileDialog openFileDialog = new OpenFileDialog();
            openFileDialog.InitialDirectory = "D:\\ДЭ\\2026\\maket\\img";
            if (openFileDialog.ShowDialog() == true)
                product.Photopath = "\\img\\" + openFileDialog.SafeFileName;

            BitmapImage myImage = new BitmapImage();
            myImage.BeginInit();
            myImage.UriSource = new Uri(openFileDialog.FileName);
            myImage.DecodePixelWidth = 300;
            myImage.DecodePixelHeight = 200;
            myImage.EndInit();

            img.Source = myImage;
        }

        private void btnSave_Click(object sender, RoutedEventArgs e)
        {
            StringBuilder errors = new StringBuilder();

            if (product.ProductPrice < 0)
                errors.AppendLine("Стоимость товара не может быть отрицательной");
            if (product.CountInStock < 0)
                errors.AppendLine("Количество товаров не может быть отрицательным");
            if (product.Discount < 0)
                errors.AppendLine("Скидка не может быть отрицательной");

            if (errors.Length > 0)
            {
                MessageBox.Show(errors.ToString(), "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
                return;
            }
            if (string.IsNullOrEmpty(product.ProductArticul))
            {
                var lastProduct = Entities.getContext().Product.ToList().LastOrDefault();
                if (lastProduct != null)
                {
                    product.ProductArticul = "ART-" + (lastProduct.ProductId + 1).ToString();
                }
                else
                {
                    product.ProductArticul = "ART-1";
                }
                Entities.getContext().Product.Add(product);
            }

            try
            {
                Entities.getContext().SaveChanges();
                MessageBox.Show("Информация сохранена", "Информация", MessageBoxButton.OK, MessageBoxImage.Information);
                NavigationService.GoBack();
            }
            catch (Exception ex)
            {
                MessageBox.Show(ex.Message, "Ошибка", MessageBoxButton.OK, MessageBoxImage.Error);
            }
        }
    }
